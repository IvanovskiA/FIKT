#include <iostream>
using namespace std;

struct S{
    char X;
    int Z[4];
};

union U{
    char X,T;
    int Z[4];
};

int main(){
    cout << sizeof(S) << "\t";
    cout << sizeof(U);
    return 0;
}

// ovoj kod demonstira kako da se presmeta goleminata na strukturata S i unijata U koristejki
// ja funkcija sizeof. Programata ja presmetuva i ja pecati goleminata na sekoj tip koristejki
// ja funkcijata sizeof
// struktura S = char X = 1 bajt; 3 bajti za popolnuvanje; po 4 bajti za sekoj clen = 4*4=16
// struktura S = 1+3+4 = 20
// unija U = char X = 1 bajt; 4 bajti za sekoj clen = 4*4 = 16; ja zema goleminata na
// najgolemiot clen
// unija U = 16
