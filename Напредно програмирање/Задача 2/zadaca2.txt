#include <iostream>
using namespace std;

struct S{
    char X;
    int z[4];
};

union U{
    char X;
    int z[4];
};

int main(){
    cout << sizeof(S) << "\t";
    cout << sizeof(U);
    return 0;
}

// ovaj kod demostira kako da se presmeta goleminata na struktura S i unijata U koristejki ja
// funkcijata sizeof. Programata ja presmetuva i ja pecati goleminata na sekoj tip koristejki
// ja funkacija sizeof

// struct S => char X = 1 bajt; 3 bajti za popolnuvanje; po 4 bajti za sekoj clen od nizata
// struct S => 1+3+4*4
// struct S => 20

// union U => char X = 1 bajt; po 4 bajti za sekoj clen od nizata 
// union U => se zema goleminata na najgolemiot tip clen 4*4
// union U => 16

// pecati:
// 20   16
